#布尔索引
names = np.array(['Bob','Joe','Will','Bob','Will','Joe','Joe'])
data = np.random.randn(7,4) #正态分布随机数7行，行数与names的元素个数相同
print(data)
mask = names == 'Bob' #布尔数组
print(mask)
print(data[mask]) #利用布尔数组切片，取mask中为True的值所对应的行
print(data[names=='Bob',2:])
